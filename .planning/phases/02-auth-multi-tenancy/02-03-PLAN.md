         - Return token
       - `callbacks.session({ session, token })`:
         - Set `session.user.id = token.id as string`
         - Set `session.user.role = token.role as string`
         - Set `session.user.tenant_id = token.tenant_id as string`
         - Set `session.access_token = token.access_token as string`
         - Return session

       CRITICAL: Next.js 16 uses `proxy.ts`, NOT `middleware.ts`. The exported function name is `proxy`.

       ```typescript
       import { auth } from "@/lib/auth";

       export const proxy = auth((req) => {
         const { pathname } = req.nextUrl;

         // Protected routes: everything under (dashboard) group
         const isProtected = pathname.startsWith("/home") ||
                             pathname.startsWith("/agenda") ||
                             pathname.startsWith("/pacientes") ||
                             pathname.startsWith("/medicos") ||
                             pathname.startsWith("/whatsapp") ||
                             pathname.startsWith("/dashboard") ||
                             pathname.startsWith("/usuarios");

         // Unauthenticated trying to access protected route -> login (AUTH-05)
         if (!req.auth && isProtected) {
           return Response.redirect(new URL("/login", req.nextUrl.origin));
         }

         // Authenticated visiting login -> dashboard (D-09)
         if (req.auth && pathname === "/login") {
           return Response.redirect(new URL("/home", req.nextUrl.origin));
         }
       });

       export const config = {
         matcher: ["/((?!api|_next/static|_next/image|favicon.ico).*)"],
       };
       ```
