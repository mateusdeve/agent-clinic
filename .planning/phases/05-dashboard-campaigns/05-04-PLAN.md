```typescript
export type CampaignStatus = "rascunho" | "enviando" | "concluida" | "falha";
export type RecipientStatus = "pendente" | "processando" | "enviado" | "entregue" | "lido" | "falha";
export interface Campaign { id: string; nome: string; template_nome: string; status: CampaignStatus; total_recipients: number; stats: {...}; created_at: string; }
export interface CampaignDetail extends Campaign { template_id: string; filtros: Record<string, string>; enviado_at: string | null; }
export interface CampaignRecipient { id: string; phone: string; patient_nome: string; status: RecipientStatus; erro: string | null; sent_at: string | null; }
export interface SegmentPreview { count: number; sample: { id: string; nome: string; phone: string }[]; }
export interface MessageTemplate { id: string; nome: string; corpo: string; variaveis_usadas: string[]; ... }
export interface PaginatedResponse<T> { items: T[]; total: number; page: number; per_page: number; }
```

Backend API (from Plan 01):
- GET /api/campaigns — PaginatedResponse<Campaign>
- POST /api/campaigns — body: {nome, template_id, filtros} — returns Campaign
- GET /api/campaigns/{id} — returns CampaignDetail
- GET /api/campaigns/{id}/recipients — PaginatedResponse<CampaignRecipient>
- GET /api/campaigns/preview-segment?especialidade=X&ultimo_agendamento_range=Y&status_paciente=Z — returns SegmentPreview
- GET /api/templates — PaginatedResponse<MessageTemplate>
- POST /api/conversations/{phone}/send-template — body: {template_id, variaveis} — returns {status: "sent"}
